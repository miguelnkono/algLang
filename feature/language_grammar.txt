This is the grammar of our language :
program         -> declaration * EOF ;
declaration     -> varDecl | statement ;                    // will have to fix this later, declaring variables at the top of the script is not allowed.
statement       -> exprStmt | printStmt ;
exprStmt        -> expr ";" ;
printStmt       -> "Ecrire (" expr ");" ;
expr            -> equality ;
equality        -> comparison (("!=" | "==") comparison)* ;
comparison      -> term ((">" | ">=" | "<" | "<=") term)* ;
term            -> factory (("+"| "-") factor)* ;
factor          -> unary (("/"| "*") unary)* ;
unary           -> ("!"| "-") unary | primary ;
primary         -> NUMBER | STRING | "vrai" | "faux" | "nil" | "(" expr ")" | IDENTIFIER ;


algoStmt        -> "Algorithme :" declaration ;

varDecl         -> "variable" IDENTIFIER ("<-" expression )? ";" ;      // will have to fix later, variable initialisation is not allowed in the algo language and the way variable are define is in a variable definition block.
